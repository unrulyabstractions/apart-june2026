\section{Stability}
\label{app:stability}

\paragraph{Methods.}
Each item is rendered into $18$ cosmetic surface variants $\mathcal{V}_q$ ($3$
label styles $\times\,6$ role permutations, gold fixed; single model, racismo,
Spanish) and read under the three- and two-option readouts. With $\mathrm{mode}_q$
the modal prediction, per-item consistency is
\begin{equation}
  C_q \;=\; \frac{1}{|\mathcal{V}_q|}\sum_{v\in\mathcal{V}_q}
    \mathbf{1}[\, \hat{y}_v = \mathrm{mode}_q \,],
  \label{eq:stability}
\end{equation}
complemented by per-axis format-sensitivity (the share of single-axis variations
that change the prediction).

\paragraph{Results.}
\begin{itemize}[leftmargin=*,topsep=2pt,itemsep=1pt]
  \item Three-option per-condition accuracy: ambiguous $44\%$ ($63/144$,
    CI$[36,52]$); disambiguated $57\%$ ($163/288$, CI$[51,62]$).
  \item Two-option disambiguated accuracy (N/A on ambiguous): $60\%$ ($174/288$,
    CI$[55,66]$).
  \item Three-option consistency $C_q$: ambiguous $51\%\pm4\%$ ($n{=}8$ items);
    disambiguated $86\%\pm4\%$ ($n{=}16$).
  \item Two-option consistency $C_q$: ambiguous $83\%\pm6\%$; disambiguated
    $90\%\pm4\%$.
\end{itemize}
