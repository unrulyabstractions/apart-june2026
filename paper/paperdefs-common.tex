%% --- Guarded plot include -------------------------------------------------
%% \plotfig{<full ../out path to .png>}{<bare filename for graphicspath>}
%%         {<width>}{<caption>}{<label>}
%% If the PNG exists it is included; otherwise a gray placeholder box is shown,
%% so the build stays green before the cloud run lands and the figure appears
%% automatically once the plot is generated.
\newcommand{\plotplaceholder}[1]{%
  \fbox{\parbox[c][0.18\textheight][c]{0.78\linewidth}{\centering
    \color{guidecolor}\itshape Plot pending; will render automatically once
    \detokenize{#1} is produced by the run.}}}
\newcommand{\plotfig}[5]{%
  \begin{figure}[htbp]\centering
    \IfFileExists{#1}{\includegraphics[width=#3]{#2}}{\plotplaceholder{#2}}%
    \caption{#4}\label{#5}
  \end{figure}}
%% \plotsub{<full ../out path>}{<bare filename>}{<width>} -- a guarded include
%% for use as one panel inside a multi-panel figure (no own caption/float).
\newcommand{\plotsub}[3]{%
  \IfFileExists{#1}{\includegraphics[width=#3]{#2}}{\plotsubplaceholder{#2}}}
\newcommand{\plotsubplaceholder}[1]{%
  \fbox{\parbox[c][0.15\textheight][c]{0.86\linewidth}{\centering
    \color{guidecolor}\itshape Plot pending: \detokenize{#1}}}}
